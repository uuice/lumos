## 组件示例

export function Callout({ type = 'info', children }) {
  const colors = { info: '#3b82f6', success: '#16a34a', warn: '#f59e0b' }
  return (
    <div style={{ border: '1px solid #e5e7eb', padding: '12px', borderRadius: '12px' }}>
      <strong style={{ color: colors[type] }}>{type.toUpperCase()}</strong>
      <div>{children}</div>
    </div>
  )
}

<Callout type="info">这是一个 MDX 内部组件，用于演示 JSX 渲染</Callout>

<Callout type="success">支持多种状态样式</Callout>
